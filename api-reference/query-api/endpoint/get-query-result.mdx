```python Python
  import os
  import requests

  # set API for authentication
  api_key = os.environ["DUNE_API_KEY"] 
  headers = {"x-dune-api-key": api_key}

  query_id = 1252207 # define the queryId
  base_url = f"https://api.dune.com/api/v1/query/{query_id}/results"
  params = {
      "params.LooksRare Wash Trading Filter": "ON", # add more params as needed
  }
  result_response = requests.request("GET", base_url, headers=headers, params=params)
```
